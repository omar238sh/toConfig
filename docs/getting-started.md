# Getting Started

This guide walks you from zero to a working generated config file in a few minutes.

---

## 1. Add `toconfig` to your project

```toml
# Cargo.toml
[dependencies]
toconfig = { path = "../toconfig" }   # local path (library is not yet on crates.io)
```

If you are creating a new binary that *uses* the library to emit configs, start with:

```bash
cargo new my-dotfiles
cd my-dotfiles
# add toconfig as a dependency (see above)
```

---

## 2. How the library works

Every configuration node is a Rust struct that implements one of three traits depending on the target tool:

| Trait | Used by |
|-------|---------|
| `Config` | Neovim, Fish, Helix, Waybar, Systemd, Fstab |
| `HyprlandConfig` | Hyprland |
| `IniConfig` | GTK, Qt |
| `FontconfigConfig` | Fontconfig |

Call `.generate()` on any node to get a `String` back. Call `.generate()` on a tree to get the whole file.

---

## 3. Your first config — Neovim `init.lua`

```rust
use toconfig::{Config, ConfigTree, LuaValue};
use toconfig::neovim::options::OptionNode;
use toconfig::neovim::keymap::{KeymapNode, MapRhs};

fn main() {
    let mut tree = ConfigTree::new().with_comment("init.lua — generated by toconfig");

    // Editor options
    tree.add(OptionNode::opt("number",         LuaValue::bool(true)));
    tree.add(OptionNode::opt("relativenumber", LuaValue::bool(true)));
    tree.add(OptionNode::opt("tabstop",        LuaValue::int(4)));

    // Keymaps
    tree.add(KeymapNode::n("<leader>w", MapRhs::cmd(":w<CR>")).desc("Save file"));
    tree.add(KeymapNode::n("<leader>q", MapRhs::cmd(":q<CR>")).desc("Quit"));

    println!("{}", tree.generate());
}
```

Output:

```lua
-- init.lua — generated by toconfig
vim.opt.number = true
vim.opt.relativenumber = true
vim.opt.tabstop = 4
vim.keymap.set('n', '<leader>w', ':w<CR>', { silent = true, noremap = true, desc = 'Save file' })
vim.keymap.set('n', '<leader>q', ':q<CR>', { silent = true, noremap = true, desc = 'Quit' })
```

---

## 4. Writing to a file

Use `ConfigOutput` to write the result to disk. It performs a diff-check — if the file already contains identical content the write is skipped.

```rust
use toconfig::output::ConfigOutput;

ConfigOutput::init_lua()       // targets ~/.config/nvim/init.lua
    .write(&tree)
    .expect("failed to write init.lua");
```

See [output.md](output.md) for the full API.

---

## 5. Validate before writing

Every node and tree exposes a `.validate()` / `.validate_all()` method. Call it before writing to catch configuration errors at runtime (instead of discovering them in the tool itself):

```rust
tree.validate_all().expect("config has validation errors");
ConfigOutput::init_lua().write(&tree)?;
```

## 6. Full end-to-end example (Neovim + Hyprland + Fish)

```rust
use toconfig::{Config, ConfigTree, LuaValue, RawLua};
use toconfig::neovim::options::{OptionNode, default_editor_options};
use toconfig::neovim::keymap::{KeymapNode, MapRhs};
use toconfig::neovim::theme::{ThemeNode, HighlightNode, HighlightAttrs, HexColor};
use toconfig::neovim::plugins::{LazyManager, Plugin, LspConfigNode, ServerConfig};
use toconfig::neovim::autocmd::{Augroup, AutocmdNode, AutocmdAction};
use toconfig::output::ConfigOutput;

use toconfig::hyprland::{HyprlandConfig, HyprlandConfigTree, Variable, EnvVar,
                          MonitorConfig, Bind, Dispatcher, AnimationsSection,
                          Bezier, Animation};
use toconfig::hyprland::exec::ExecOnce;
use toconfig::hyprland::section::Section;
use toconfig::hyprland::output::HyprlandOutput;

use toconfig::fish::config_tree::FishConfigTree;
use toconfig::fish::variable::FishVariable;
use toconfig::fish::alias::FishAlias;
use toconfig::fish::path::FishAddPath;
use toconfig::fish::prompt::FishGreeting;

fn main() -> std::io::Result<()> {
    // ── Neovim ────────────────────────────────────────────────────────────────
    let mut nvim_tree = ConfigTree::new().with_comment("init.lua — generated by toconfig");
    nvim_tree.add(OptionNode::g("mapleader", LuaValue::str(" ")));
    nvim_tree.add(
        LazyManager::new()
            .plugin(Plugin::new("folke/tokyonight.nvim").priority(1000).lazy(false))
            .plugin(Plugin::new("neovim/nvim-lspconfig"))
            .plugin(Plugin::new("nvim-treesitter/nvim-treesitter").build(":TSUpdate")),
    );
    nvim_tree.add(default_editor_options());
    nvim_tree.add(ThemeNode::new("tokyonight-night"));
    nvim_tree.add(KeymapNode::n("<leader>w", MapRhs::cmd(":w<CR>")).desc("Save"));
    nvim_tree.add(
        Augroup::new("UserGroup")
            .add(AutocmdNode::on_buf_write(
                AutocmdAction::callback("function() vim.lsp.buf.format({ async = true }) end"),
            ).desc("Format on save"))
    );
    nvim_tree.add(
        LspConfigNode::new()
            .server(ServerConfig::new("rust_analyzer").filetypes(&["rust"]))
            .server(ServerConfig::new("lua_ls")),
    );
    nvim_tree.validate_all().expect("nvim validation failed");
    ConfigOutput::init_lua().write(&nvim_tree)?;

    // ── Hyprland ──────────────────────────────────────────────────────────────
    let mut hypr_tree = HyprlandConfigTree::new();
    hypr_tree.add(Variable::new("terminal", "kitty"));
    hypr_tree.add(EnvVar::xcursor_size(24));
    hypr_tree.add(MonitorConfig::new("eDP-1", "1920x1080@60", "0x0", 1.0));
    hypr_tree.add(ExecOnce::new("waybar"));
    hypr_tree.add(Bind::new("SUPER", "Return", Dispatcher::Exec("$terminal".into())));
    hypr_tree.add(Bind::new("SUPER", "Q", Dispatcher::KillActive));
    hypr_tree.add(
        Section::new("general")
            .pair("gaps_in", "5")
            .pair("gaps_out", "20")
            .pair("border_size", "2"),
    );
    HyprlandOutput::hyprland_conf().write(&hypr_tree)?;

    // ── Fish ──────────────────────────────────────────────────────────────────
    let fish_tree = FishConfigTree::new()
        .with_comment("config.fish — generated by toconfig")
        .add(FishGreeting::disabled())
        .add(FishVariable::env("EDITOR", "nvim"))
        .add(FishAddPath::new(&["$HOME/.local/bin", "$HOME/.cargo/bin"]))
        .add(FishAlias::new("ll", "ls -la"));
    fish_tree.validate_all().expect("fish validation failed");
    let fish_path = format!("{}/.config/fish/config.fish", std::env::var("HOME").unwrap());
    ConfigOutput::at_path(&fish_path).write(&fish_tree)?;

    Ok(())
}
```
